CREATE NETWORK IF NEEDED:

docker network create task-network


POSTGRE SQL CONTAINER:

docker run -d
  --name task-postgres-local
  -e POSTGRES_DB=task-db
  -e POSTGRES_USER=task-user
  -e POSTGRES_PASSWORD=task-pw
  -p 5432:5432
  --network task-network
  postgres:alpine


CREATE APP IMAGE AND CONTAINER

  docker build -t task-app .

  docker run --name task-app -p 5000:5000 --network task-network -t task-app
